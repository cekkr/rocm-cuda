2023template<typename UnaryFunction, class T>
2024static __inline__ __host__ CUDART_DEVICE cudaError_t cudaOccupancyMaxPotentialBlockSizeVariableSMemWithFlags(
2025 int *minGridSize,
2026 int *blockSize,
2027 T func,
2028 UnaryFunction blockSizeToDynamicSMemSize,
2029 int blockSizeLimit = 0,
2030 unsigned int flags = 0)
2031{
2032 cudaError_t status;
2033
2034 // Device and function properties
2035 int device;
2036 struct cudaFuncAttributes attr;
2037
2038 // Limits
2039 int maxThreadsPerMultiProcessor;
2040 int warpSize;
2041 int devMaxThreadsPerBlock;
2042 int multiProcessorCount;
2043 int funcMaxThreadsPerBlock;
2044 int occupancyLimit;
2045 int granularity;
2046
2047 // Recorded maximum
2048 int maxBlockSize = 0;
2049 int numBlocks = 0;
2050 int maxOccupancy = 0;
2051
2052 // Temporary
2053 int blockSizeToTryAligned;
2054 int blockSizeToTry;
2055 int blockSizeLimitAligned;
2056 int occupancyInBlocks;
2057 int occupancyInThreads;
2058 size_t dynamicSMemSize;
2059
2061 // Check user input
2063
2064 if (!minGridSize || !blockSize || !func) {
2065 return cudaErrorInvalidValue;
2066 }
2067
2069 // Obtain device and function properties
2071
2072 status = ::cudaGetDevice(&device);
2073 if (status != cudaSuccess) {
2074 return status;
2075 }
2076
2077 status = cudaDeviceGetAttribute(
2078 &maxThreadsPerMultiProcessor,
2079 cudaDevAttrMaxThreadsPerMultiProcessor,
2080 device);
2081 if (status != cudaSuccess) {
2082 return status;
2083 }
2084
2085 status = cudaDeviceGetAttribute(
2086 &warpSize,
2087 cudaDevAttrWarpSize,
2088 device);
2089 if (status != cudaSuccess) {
2090 return status;
2091 }
2092
2093 status = cudaDeviceGetAttribute(
2094 &devMaxThreadsPerBlock,
2095 cudaDevAttrMaxThreadsPerBlock,
2096 device);
2097 if (status != cudaSuccess) {
2098 return status;
2099 }
2100
2101 status = cudaDeviceGetAttribute(

2104 device);

2109 status = cudaFuncGetAttributes(&attr, func);
2110 if (status != cudaSuccess) {
2111 return status;

2113
2114 funcMaxThreadsPerBlock = attr.maxThreadsPerBlock;
2115
2117 // Try each block size, and pick the block size with maximum occupancy
2119
2120 occupancyLimit = maxThreadsPerMultiProcessor;
2121 granularity = warpSize;
2122
2123 if (blockSizeLimit == 0) {
2124 blockSizeLimit = devMaxThreadsPerBlock;
2125 }
2126
2127 if (devMaxThreadsPerBlock < blockSizeLimit) {
2128 blockSizeLimit = devMaxThreadsPerBlock;

2131 if (funcMaxThreadsPerBlock < blockSizeLimit) {

2139 // blockSizeLimitAligned could be greater than blockSizeLimit
2140 //
2141 if (blockSizeLimit < blockSizeToTryAligned) {
2142 blockSizeToTry = blockSizeLimit;
2143 } else {
2144 blockSizeToTry = blockSizeToTryAligned;
2145 }
2146
2147 dynamicSMemSize = blockSizeToDynamicSMemSize(blockSizeToTry);
2148
2149 status = cudaOccupancyMaxActiveBlocksPerMultiprocessorWithFlags(

2151 func,
2152 blockSizeToTry,
2153 dynamicSMemSize,
2154 flags);
2155
2156 if (status != cudaSuccess) {
2157 return status;
2158 }
2159
2160 occupancyInThreads = blockSizeToTry * occupancyInBlocks;
2161

2163 maxBlockSize = blockSizeToTry;
2164 numBlocks = occupancyInBlocks;
2165 maxOccupancy = occupancyInThreads;
2166 }
2167
2168 // Early out if we have reached the maximum
2169 //
2170 if (occupancyLimit == maxOccupancy) {
2171 break;
2172 }
2173 }
2174
2176 // Return best available
2178
2179 // Suggested min grid size to achieve a full machine launch
2180 //
2181 *minGridSize = numBlocks * multiProcessorCount;
2182 *blockSize = maxBlockSize;

2184 return status;
